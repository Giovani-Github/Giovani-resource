
CREATE TABLESPACE ZJUSER
    LOGGING DATAFILE  'D:\giovani-application\oracle\oradata\mydata\ZJUSER.DBF'
    SIZE  1000M
    REUSE AUTOEXTEND ON 
    NEXT  100M
    MAXSIZE UNLIMITED EXTENT 
    MANAGEMENT LOCAL;

CREATE TEMPORARY TABLESPACE ZJUSER_TEMP
    TEMPFILE  'D:\giovani-application\oracle\oradata\mydata\ZJUSER_TEMP.DBF'
    SIZE  50M
    REUSE AUTOEXTEND ON 
    NEXT  100M
    MAXSIZE UNLIMITED EXTENT 
    MANAGEMENT LOCAL;

    
CREATE USER ZJUSER  
    PROFILE DEFAULT 
    IDENTIFIED BY "123456"
    DEFAULT TABLESPACE   ZJUSER
    TEMPORARY TABLESPACE ZJUSER_TEMP;


GRANT CONNECT TO ZJUSER;

GRANT DBA TO ZJUSER;

GRANT SELECT ON DBA_OBJECTS TO ZJUSER;

GRANT SELECT ON USER_TAB_COLUMNS TO ZJUSER;

GRANT SELECT ON USER_TABLES TO ZJUSER;

GRANT CREATE ANY TABLE TO ZJUSER;
